### django cmd's ->
# 1. Delete old database
del db.sqlite3

# 2. Delete old migration files (keep __init__.py in each folder)
# Delete: apps/accounts/migrations/0001_*.py (if exists)
# Delete: apps/evidence/migrations/0001_*.py (if exists)
# Delete: apps/blockchain/migrations/0001_*.py (if exists)

# 3. Fresh migrations
python manage.py makemigrations --settings=core.settings.local

# 4. Apply
python manage.py migrate --settings=core.settings.local

# 5. Create superuser
python manage.py createsuperuser --settings=core.settings.local

# 6. Run server
python manage.py runserver --settings=core.settings.local
